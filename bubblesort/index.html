   public static void sorted (int [] alfa) { 
      int i,j; 
      for ( i=alfa.length-1; i>0; i--){ 
         for( j=0; j < i; j++) { 
            if (alfa[j] > alfa[j+1]) { 
               int temp = alfa [j]; 
               alfa [j] = alfa [j+1]; 
               alfa[j+1] = temp; 
            } 
         } 
      } 
   }
